pytest>=8.0.0
pytest-asyncio>=0.23.0
# httpx provides AsyncClient; already pinned in requirements.txt — listed

httpx>=0.27.0
# testcontainers spins up an ephemeral Postgres in Docker. Used when
# TEST_DATABASE_URL is not set. If Docker isn't available, integration
# tests are skipped (see tests/conftest.py).
testcontainers[postgres]>=4.7.0
# Coverage is optional but cheap to install.
pytest-cov>=5.0.0
# Per-test timeout. A single hung test (e.g. a DB call that blocks on a
# lock) would otherwise silently eat the whole job timeout and report as
# "cancelled" with no traceback. With this, the test is killed at the cap
# and pytest prints exactly where it hung. thread method is used in CI.
pytest-timeout>=2.3.0
# Surfaces each test failure as a GitHub `::error::` annotation (auto-active
# when GITHUB_ACTIONS=true) so failures show inline on the PR diff and are
# readable via the check-runs annotations API without needing raw log access.
pytest-github-actions-annotate-failures>=0.2.0
